1). Import http from node to create basic node server
example :- 
    let http = require('http')
    let server = http.createServer((req, res) => {
        res.end("<h1>Hello</h1>")
    })
    server.listen(8080)

2). We can set headers in request to let browser the the type of our content or any other info
example :- 
    let server = http.createServer((req, res) => {
        res.setHeader('Content-Type','text/html')
        res.end("Hello")

        res.setHeader('Content-Type','application/json')
        res.end(JSON.stringify(someData))
    })
